% Set default caption font type
\DeclareCaptionFont{figurefont}{\zihao{5}\bfseries\songti}
\captionsetup[figure]{font=figurefont}
\captionsetup[table]{font=figurefont}

% `position` doesn't actually set the caption position,
% It only determines weather the `skip` inserted above/below `caption`
\captionsetup{position=bottom}

% Use whitespace as caption separators
\captionsetup[figure]{labelsep=space}
\captionsetup[table]{labelsep=space}

%% Uncomment the following lines to use dual-language caption
%% E.g.: https://github.com/TheNetAdmin/zjuthesis/issues/181
\usepackage[list=off]{bicaption}
\DeclareCaptionOption{bi-second}[]{%
    \def\tablename{Table}%
    \def\figurename{Figure}%
}
\captionsetup[bi-second]{bi-second}

% Put bilingual figure/table entries in the same list item:
% Chinese title on the first line and English title on the second line.
\makeatletter
\renewcommand\bicaption@addcontentsline[3]{%
    \begingroup
        \let\caption@addsubcontentslines\@gobble
        \caption@setoptions{bi}%
        \caption@setoptions{bi-first}%
        \caption@ifcontentsline{#3}{%
            \edef\bicaption@tempa{\@nameuse{ext@#2}}%
            \def\bicaption@tempb{lof}%
            \ifx\bicaption@tempa\bicaption@tempb
                \bicaption@addcontentsline@ORI{#2}{#3\protect\\\protect\hspace*{-\cftfignumwidth}Figure~\@nameuse{the#2}\enspace #1}%
            \else
                \bicaption@addcontentsline@ORI{#2}{#3\protect\\\protect\hspace*{-\cfttabnumwidth}Table~\@nameuse{the#2}\enspace #1}%
            \fi
        }%
    \endgroup
    \caption@addsubcontentslines{#2}%
}
\makeatother
